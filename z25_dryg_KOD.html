/*
  Autor: Drygalski
  Zadanie 25 - Obliczanie ilości atomów pierwiastka promieniotwórczego po czasie t
*/

function zadanie25() {
  var No_dry = parseFloat(prompt("Podaj początkową ilość atomów No:"));
  var t_dry = parseFloat(prompt("Podaj czas t (w latach):"));
  var T12_dry = parseFloat(prompt("Podaj okres połowicznego zaniku T1/2 (w latach):"));

  if (isNaN(No_dry) || isNaN(t_dry) || isNaN(T12_dry) || No_dry <= 0 || t_dry < 0 || T12_dry <= 0) {
    alert("Proszę podać poprawne liczby. No > 0, t >= 0, T1/2 > 0");
    return;
  }

  var lambda_dry = Math.log(2) / T12_dry;
  var N_dry = No_dry * Math.pow(Math.E, -lambda_dry * t_dry);

  var output = "";

  output += `<p style="font-size:18px; color:#003366;">Wczytane dane:</p>`;
  output += `<p style="font-size:16px; color:#004d00;">No = ${No_dry.toFixed(2)} [sztuk]</p>`;
  output += `<p style="font-size:16px; color:#004d00;">t = ${t_dry.toFixed(2)} [lat]</p>`;
  output += `<p style="font-size:16px; color:#004d00;">T1/2 = ${T12_dry.toFixed(2)} [lat]</p>`;

  output += `<hr>`;

  output += `<p style="font-size:18px; color:#800000;">Wyniki obliczeń:</p>`;
  output += `<p style="font-size:16px; color:#990000;">Stała rozpadu &lambda; = ${lambda_dry.toFixed(7)} [1/rok]</p>`;
  output += `<p style="font-size:16px; color:#990000;">Ilość atomów po czasie t: N = ${N_dry.toFixed(2)} [sztuk]</p>`;

  document.getElementById('wyniki_dry').innerHTML = output;
}
</script>
