<script>
function Pole_dry(a_dry, b_dry) {
  return a_dry * b_dry;
}

function obwod_dry(a_dry, b_dry) {
  return 2 * (a_dry + b_dry);
}

function licz() {
  var a_dry = parseFloat(document.getElementById('a_dana_dry').value);
  var b_dry = parseFloat(document.getElementById('b_dana_dry').value);

  if (isNaN(a_dry) || isNaN(b_dry)) {
    alert('Proszę podać poprawne liczby!');
    return;
  }

  var pole = Pole_dry(a_dry, b_dry);
  var obw = obwod_dry(a_dry, b_dry);

  document.getElementsByName('pole')[0].value = pole.toFixed(2);
  document.getElementsByName('obw')[0].value = obw.toFixed(2);

  document.getElementById('wyprowadz_pole_dry').innerHTML = "Pole prostokąta = " + pole.toFixed(2) + " cm²";
  document.getElementById('wyprowadz_obwod_dry').innerHTML = "Obwód prostokąta = " + obw.toFixed(2) + " cm";
}
</script>
