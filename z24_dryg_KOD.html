function zadanie24() {
  var x1_dry = parseFloat(prompt("Podaj x1 (w stopniach):"));
  var x2_dry = parseFloat(prompt("Podaj x2:"));

  if (isNaN(x1_dry) || isNaN(x2_dry)) {
    alert("Proszę podać poprawne liczby.");
    return;
  }

  var wczytane = "wczytane liczby: x1=" + x1_dry + "      x2=" + x2_dry;
  document.getElementById('wczytane_dry').innerHTML = wczytane;

  var y1 = Math.pow(x1_dry, x2_dry);
  var rad_dry = (x1_dry * Math.PI) / 180;
  var y2 = Math.sin(rad_dry);
  var y3 = Math.cos(rad_dry);

  var wyniki = "y1 = " + y1.toFixed(2) + "<br>" +
               "y2 = " + y2.toFixed(2) + "<br>" +
               "y3 = " + y3.toFixed(2);
  document.getElementById('wyniki_dry').innerHTML = wyniki;

  var dodatkowe = "x1 w stopniach = " + x1_dry.toFixed(2) + " [stopniach]<br>" +
                  "x1 w radianach = " + rad_dry.toFixed(2) + " [radianach]";
  document.getElementById('dodatkowe_dry').innerHTML = dodatkowe;
}
</script>
